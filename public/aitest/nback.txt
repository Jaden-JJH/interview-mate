import React, { useState, useEffect, useCallback, useRef } from 'react';

const NBackGame = () => {
  const shapeGroups = [
    [
      { id: 'triangle', name: '삼각형', symbol: '▲' },
      { id: 'circle', name: '원', symbol: '●' },
      { id: 'square', name: '사각형', symbol: '■' }
    ],
    [
      { id: 'trapezoid', name: '사다리꼴', symbol: '▼' },
      { id: 'hourglass', name: '모래시계', symbol: '⧗' },
      { id: 'diamond', name: '다이아몬드', symbol: '♦' }
    ],
    [
      { id: 'rotated-diamond', name: '회전 다이아몬드', symbol: '◆' },
      { id: 'butterfly', name: '나비', symbol: '⌘' },
      { id: 'star', name: '별', symbol: '★' }
    ],
    [
      { id: 'checkerboard', name: '체크보드', symbol: '▣' },
      { id: 'double-down-triangle', name: '이중 역삼각형', symbol: '▼' },
      { id: 'pyramid', name: '피라미드', symbol: '▲' }
    ],
    [
      { id: 'arrow-double', name: '이중 화살표', symbol: '⇈' },
      { id: 'x-shape', name: 'X 모양', symbol: '✕' },
      { id: 'crown', name: '왕관', symbol: '♔' }
    ]
  ];

  const [gameState, setGameState] = useState('setup');
  const [selectedGroupIndex, setSelectedGroupIndex] = useState(0);
  const [displayTime, setDisplayTime] = useState(3000);
  const [totalQuestions, setTotalQuestions] = useState(20);
  const [nBackLevel, setNBackLevel] = useState(2);
  const [useRandomShapes, setUseRandomShapes] = useState(false);
  const [actualGroupIndex, setActualGroupIndex] = useState(0);
  
  const [showHistory, setShowHistory] = useState(true);
  const [historyCount, setHistoryCount] = useState(3);
  const [showHints, setShowHints] = useState(true);
  const [intervalTime, setIntervalTime] = useState(500);
  
  const selectedShapes = shapeGroups[actualGroupIndex];
  
  const [currentTrial, setCurrentTrial] = useState(0);
  const [shapeHistory, setShapeHistory] = useState([]);
  const [currentShape, setCurrentShape] = useState(null);
  const [score, setScore] = useState(0);
  const [totalTrials, setTotalTrials] = useState(0);
  const [streak, setStreak] = useState(0);
  const [isShowingShape, setIsShowingShape] = useState(false);
  const [userAnswer, setUserAnswer] = useState(null);
  const [timerProgress, setTimerProgress] = useState(0);
  const [isInInterval, setIsInInterval] = useState(false);
  
  const gameTimer = useRef(null);
  const userAnswerRef = useRef(null);
  const progressTimer = useRef(null);

  const handleGroupSelection = (groupIndex) => {
    setSelectedGroupIndex(groupIndex);
    setActualGroupIndex(groupIndex);
    setUseRandomShapes(false);
  };

  const getNextShape = (history) => {
    const lastShape = history[history.length - 1];
    const secondLastShape = history[history.length - 2];
    
    // 직전 2개가 같은 도형인지 체크
    if (lastShape && secondLastShape && lastShape.id === secondLastShape.id) {
      // 3번 연속 방지: 반드시 다른 도형 선택
      const differentShapes = selectedShapes.filter(s => s.id !== lastShape.id);
      return differentShapes[Math.floor(Math.random() * differentShapes.length)];
    } else {
      // 자유롭게 선택
      return selectedShapes[Math.floor(Math.random() * selectedShapes.length)];
    }
  };

  const showNextTrial = () => {
    setCurrentTrial(prev => {
      const newTrial = prev + 1;
      
      setShapeHistory(prevHistory => {
        // 다음 도형 선택 (3연속 방지 로직 포함)
        const nextShape = getNextShape(prevHistory);
        const newHistory = [...prevHistory, nextShape];
        
        setCurrentShape(nextShape);
        setUserAnswer(null);
        userAnswerRef.current = null;
        setIsShowingShape(true);
        setIsInInterval(false);
        setTimerProgress(0);

        const startTime = Date.now();
        progressTimer.current = setInterval(() => {
          const elapsed = Date.now() - startTime;
          const progress = Math.min((elapsed / displayTime) * 100, 100);
          setTimerProgress(progress);
          
          if (progress >= 100) {
            clearInterval(progressTimer.current);
          }
        }, 16);

        gameTimer.current = setTimeout(() => {
          setIsShowingShape(false);
          setTimerProgress(100);
          
          if (progressTimer.current) {
            clearInterval(progressTimer.current);
          }
          
          const userResponse = userAnswerRef.current || 'timeout';
          let expectedResponse = 'none';

          if (nBackLevel === 2) {
            if (newTrial === 1 || newTrial === 2) {
              expectedResponse = 'none';
            } else {
              const curr = newHistory[newTrial - 1];
              const twoBack = newHistory[newTrial - 3];
              expectedResponse = curr?.id === twoBack?.id ? '2back' : 'none';
            }
          } else {
            if (newTrial === 1 || newTrial === 2) {
              expectedResponse = 'none';
            } else if (newTrial === 3) {
              const curr = newHistory[2];
              const twoBack = newHistory[0];
              expectedResponse = curr?.id === twoBack?.id ? '2back' : 'none';
            } else {
              const curr = newHistory[newTrial - 1];
              const twoBack = newHistory[newTrial - 3];
              const threeBack = newHistory[newTrial - 4];

              if (curr?.id === threeBack?.id) {
                expectedResponse = '3back';
              } else if (curr?.id === twoBack?.id) {
                expectedResponse = '2back';
              } else {
                expectedResponse = 'none';
              }
            }
          }

          if (newTrial >= 3) {
            const isCorrect = userResponse !== 'timeout' && userResponse === expectedResponse;
            setTotalTrials(prevTotal => prevTotal + 1);
            
            if (isCorrect) {
              setScore(prevScore => prevScore + 1);
              setStreak(prevStreak => prevStreak + 1);
            } else {
              setStreak(0);
            }
          }

          if (newTrial < totalQuestions) {
            setIsInInterval(true);
            setTimeout(() => {
              showNextTrial();
            }, intervalTime);
          } else {
            setGameState('finished');
          }
        }, displayTime);

        return newHistory;
      });
      
      return newTrial;
    });
  };

  const handleKeyPress = useCallback((event) => {
    if (!isShowingShape || gameState !== 'playing' || userAnswer !== null) return;

    let userResponse = 'none';

    if (nBackLevel === 2) {
      if (event.code === 'ArrowLeft') {
        userResponse = '2back';
      } else if (event.code === 'Space') {
        userResponse = 'none';
        event.preventDefault();
      } else {
        return;
      }
    } else {
      if (event.code === 'ArrowLeft') {
        userResponse = '2back';
      } else if (event.code === 'ArrowRight') {
        userResponse = '3back';
      } else if (event.code === 'Space') {
        userResponse = 'none';
        event.preventDefault();
      } else {
        return;
      }
    }

    setUserAnswer(userResponse);
    userAnswerRef.current = userResponse;
  }, [isShowingShape, gameState, userAnswer, nBackLevel]);

  useEffect(() => {
    document.addEventListener('keydown', handleKeyPress);
    return () => {
      document.removeEventListener('keydown', handleKeyPress);
    };
  }, [handleKeyPress]);

  const startGame = () => {
    if (useRandomShapes) {
      const randomIndex = Math.floor(Math.random() * shapeGroups.length);
      setActualGroupIndex(randomIndex);
    } else {
      setActualGroupIndex(selectedGroupIndex);
    }
    
    setGameState('playing');
    setCurrentTrial(0);
    setShapeHistory([]);
    setScore(0);
    setTotalTrials(0);
    setStreak(0);
    setUserAnswer(null);
    userAnswerRef.current = null;
    setIsShowingShape(false);
    setIsInInterval(false);
    setTimerProgress(0);
    
    if (gameTimer.current) clearTimeout(gameTimer.current);
    if (progressTimer.current) clearInterval(progressTimer.current);
    
    setTimeout(() => {
      showNextTrial();
    }, 500);
  };

  const resetGame = () => {
    setGameState('setup');
    userAnswerRef.current = null;
    setTimerProgress(0);
    setIsInInterval(false);
    if (gameTimer.current) clearTimeout(gameTimer.current);
    if (progressTimer.current) clearInterval(progressTimer.current);
  };

  useEffect(() => {
    return () => {
      if (gameTimer.current) clearTimeout(gameTimer.current);
      if (progressTimer.current) clearInterval(progressTimer.current);
    };
  }, []);

  return (
    <div className="min-h-screen bg-gradient-to-br from-gray-50 to-gray-100 p-8">
      <div className="max-w-5xl mx-auto">
        <div className="text-center mb-8">
          <h1 className="text-5xl font-bold text-gray-800 mb-2">
            N-Back 도형 게임
          </h1>
          <p className="text-gray-600">작업 기억 훈련 게임</p>
        </div>

        {gameState === 'setup' && (
          <div className="bg-white rounded-2xl shadow-xl p-8">
            
            <div className="mb-8 pb-8 border-b border-gray-200">
              <h3 className="text-xl font-semibold mb-4 text-gray-800">난이도 설정</h3>
              <div className="flex gap-3">
                <button
                  onClick={() => setNBackLevel(2)}
                  className={`flex-1 py-4 rounded-xl font-semibold transition-all ${
                    nBackLevel === 2 
                      ? 'bg-blue-600 text-white shadow-lg scale-105' 
                      : 'bg-gray-100 text-gray-700 hover:bg-gray-200'
                  }`}
                >
                  2-Back
                </button>
                <button
                  onClick={() => setNBackLevel(3)}
                  className={`flex-1 py-4 rounded-xl font-semibold transition-all ${
                    nBackLevel === 3 
                      ? 'bg-blue-600 text-white shadow-lg scale-105' 
                      : 'bg-gray-100 text-gray-700 hover:bg-gray-200'
                  }`}
                >
                  3-Back
                </button>
              </div>
            </div>

            <div className="mb-8 pb-8 border-b border-gray-200">
              <h3 className="text-xl font-semibold mb-4 text-gray-800">도형 설정</h3>
              
              <div className="mb-4">
                <label className="flex items-center gap-3 p-4 bg-purple-50 rounded-xl border-2 border-purple-200 cursor-pointer hover:bg-purple-100 transition">
                  <input
                    type="checkbox"
                    checked={useRandomShapes}
                    onChange={(e) => setUseRandomShapes(e.target.checked)}
                    className="w-5 h-5 accent-purple-600"
                  />
                  <div className="flex-1">
                    <span className="font-semibold text-purple-900">랜덤 그룹 선택</span>
                    <p className="text-sm text-purple-700 mt-1">5개 그룹 중 무작위로 한 그룹 선택 (어떤 그룹인지 모름)</p>
                  </div>
                </label>
              </div>
              
              {!useRandomShapes && (
                <>
                  <div className="grid grid-cols-5 gap-2 mb-4">
                    {shapeGroups.map((group, groupIndex) => (
                      <button
                        key={groupIndex}
                        onClick={() => handleGroupSelection(groupIndex)}
                        className={`p-3 rounded-lg border-2 transition-all ${
                          selectedGroupIndex === groupIndex
                            ? 'border-blue-500 bg-blue-50 shadow-md'
                            : 'border-gray-200 bg-white hover:border-gray-300'
                        }`}
                      >
                        <div className="text-xs font-medium mb-2 text-gray-600">그룹 {groupIndex + 1}</div>
                        <div className="flex justify-center gap-1">
                          {group.map((shape, shapeIndex) => (
                            <div key={shapeIndex} className="text-teal-600 text-lg">
                              {shape.symbol}
                            </div>
                          ))}
                        </div>
                      </button>
                    ))}
                  </div>
                  
                  <div className="grid grid-cols-3 gap-3 bg-gray-50 p-4 rounded-xl">
                    {shapeGroups[selectedGroupIndex].map((shape, index) => (
                      <div key={index} className="bg-white rounded-lg p-3 text-center shadow-sm">
                        <div className="text-4xl text-teal-600 mb-1">{shape.symbol}</div>
                        <div className="text-xs text-gray-600">{shape.name}</div>
                      </div>
                    ))}
                  </div>
                </>
              )}
              
              {useRandomShapes && (
                <div className="bg-gray-50 p-4 rounded-xl text-center text-gray-600">
                  <div className="text-sm">게임 시작 시 <strong className="text-purple-600">5개 그룹 중 1개</strong>가 무작위로 선택됩니다</div>
                  <div className="text-xs text-gray-500 mt-1">어떤 그룹인지 미리 알 수 없습니다</div>
                </div>
              )}
            </div>

            <div className="mb-8 pb-8 border-b border-gray-200">
              <h3 className="text-xl font-semibold mb-4 text-gray-800">연습 모드</h3>
              
              <div className="space-y-4">
                <label className="flex items-start gap-3 cursor-pointer">
                  <input
                    type="checkbox"
                    checked={showHistory}
                    onChange={(e) => setShowHistory(e.target.checked)}
                    className="w-5 h-5 mt-1 accent-blue-600"
                  />
                  <div className="flex-1">
                    <div className="font-medium text-gray-800">이전 도형 기록 표시</div>
                    <p className="text-sm text-gray-600">이전 도형들을 화면에 표시</p>
                    {showHistory && (
                      <div className="mt-2">
                        <input
                          type="range"
                          min="2"
                          max="5"
                          value={historyCount}
                          onChange={(e) => setHistoryCount(parseInt(e.target.value))}
                          className="w-full h-2 bg-gray-200 rounded-lg"
                        />
                        <div className="text-xs text-gray-500 mt-1">표시 개수: {historyCount}개</div>
                      </div>
                    )}
                  </div>
                </label>

                <label className="flex items-start gap-3 cursor-pointer">
                  <input
                    type="checkbox"
                    checked={showHints}
                    onChange={(e) => setShowHints(e.target.checked)}
                    className="w-5 h-5 mt-1 accent-blue-600"
                  />
                  <div className="flex-1">
                    <div className="font-medium text-gray-800">비교 대상 강조</div>
                    <p className="text-sm text-gray-600">{nBackLevel}번째 전 도형을 색상으로 강조</p>
                  </div>
                </label>
              </div>
            </div>

            <div className="mb-8 pb-8 border-b border-gray-200">
              <h3 className="text-xl font-semibold mb-4 text-gray-800">타이밍 설정</h3>
              
              <div className="space-y-4">
                <div>
                  <div className="flex justify-between mb-2">
                    <span className="text-gray-700">도형 표시 시간</span>
                    <span className="font-semibold text-gray-900">{displayTime / 1000}초</span>
                  </div>
                  <input
                    type="range"
                    min="500"
                    max="5000"
                    step="100"
                    value={displayTime}
                    onChange={(e) => setDisplayTime(parseInt(e.target.value))}
                    className="w-full h-2 bg-gray-200 rounded-lg"
                  />
                </div>

                <div>
                  <div className="flex justify-between mb-2">
                    <span className="text-gray-700">도형 사이 간격</span>
                    <span className="font-semibold text-gray-900">{intervalTime / 1000}초</span>
                  </div>
                  <input
                    type="range"
                    min="0"
                    max="2000"
                    step="100"
                    value={intervalTime}
                    onChange={(e) => setIntervalTime(parseInt(e.target.value))}
                    className="w-full h-2 bg-gray-200 rounded-lg"
                  />
                </div>

                <div>
                  <div className="flex justify-between mb-2">
                    <span className="text-gray-700">문제 수</span>
                    <span className="font-semibold text-gray-900">{totalQuestions}문제</span>
                  </div>
                  <input
                    type="range"
                    min="10"
                    max="100"
                    step="5"
                    value={totalQuestions}
                    onChange={(e) => setTotalQuestions(parseInt(e.target.value))}
                    className="w-full h-2 bg-gray-200 rounded-lg"
                  />
                </div>
              </div>
            </div>

            <div className="mb-6 bg-blue-50 rounded-xl p-4">
              <h3 className="font-semibold text-blue-900 mb-3">키보드 조작법</h3>
              <div className="space-y-2 text-sm text-blue-800">
                <div className="flex items-center gap-2">
                  <span className="bg-white px-3 py-1 rounded font-mono text-xs">SPACE</span>
                  <span>다름</span>
                </div>
                <div className="flex items-center gap-2">
                  <span className="bg-blue-600 text-white px-3 py-1 rounded font-mono text-xs">←</span>
                  <span>2번째 전과 같음</span>
                </div>
                {nBackLevel === 3 && (
                  <div className="flex items-center gap-2">
                    <span className="bg-purple-600 text-white px-3 py-1 rounded font-mono text-xs">→</span>
                    <span>3번째 전과 같음</span>
                  </div>
                )}
              </div>
            </div>

            <button
              onClick={startGame}
              className="w-full bg-gradient-to-r from-blue-600 to-blue-700 text-white py-4 rounded-xl text-xl font-bold hover:from-blue-700 hover:to-blue-800 transition-all shadow-lg hover:shadow-xl"
            >
              게임 시작
            </button>
          </div>
        )}

        {gameState === 'playing' && (
          <div className="bg-white rounded-2xl shadow-xl p-8">
            
            <div className="flex justify-between items-center mb-6 pb-4 border-b border-gray-200">
              <div className="text-sm">
                <div className="text-gray-500">문항</div>
                <div className="text-xl font-bold text-gray-800">{currentTrial} / {totalQuestions}</div>
              </div>
              <div className="text-sm text-center">
                <div className="text-gray-500">정답률</div>
                <div className="text-xl font-bold text-blue-600">
                  {totalTrials > 0 ? Math.round((score / totalTrials) * 100) : 0}%
                </div>
              </div>
              <div className="text-sm text-center">
                <div className="text-gray-500">연속</div>
                <div className="text-xl font-bold text-green-600">{streak}</div>
              </div>
              <button
                onClick={resetGame}
                className="bg-gray-100 text-gray-700 px-4 py-2 rounded-lg font-medium hover:bg-gray-200 transition"
              >
                나가기
              </button>
            </div>

            <div className="mb-8">
              <div className="w-full bg-gray-100 rounded-full h-3 overflow-hidden">
                <div 
                  className={`h-3 rounded-full transition-all duration-75 ease-linear ${
                    isShowingShape ? 'bg-gradient-to-r from-blue-500 to-blue-600' : 'bg-transparent'
                  }`}
                  style={{ width: `${timerProgress}%` }}
                ></div>
              </div>
            </div>

            <div className="flex items-start justify-center gap-8">
              
              {showHistory && (
                <div className="flex-shrink-0">
                  <div className="text-xs font-medium text-gray-500 mb-3 text-center">이전 도형</div>
                  <div className="flex flex-col gap-2">
                    {Array.from({ length: Math.min(historyCount, currentTrial - 1) }).map((_, idx) => {
                      const historyIdx = shapeHistory.length - 2 - idx;
                      if (historyIdx < 0) return null;
                      const shape = shapeHistory[historyIdx];
                      const positionFromCurrent = idx + 2;
                      
                      const shouldHighlight = showHints && (
                        (nBackLevel === 2 && positionFromCurrent === 2) ||
                        (nBackLevel === 3 && (positionFromCurrent === 2 || positionFromCurrent === 3))
                      );
                      
                      return (
                        <div 
                          key={idx} 
                          className={`rounded-lg p-2 transition-all ${
                            shouldHighlight 
                              ? positionFromCurrent === 2 
                                ? 'bg-blue-100 border-2 border-blue-500' 
                                : 'bg-purple-100 border-2 border-purple-500'
                              : 'bg-gray-50 border-2 border-gray-200'
                          }`}
                        >
                          <div className="text-2xs text-center mb-1 font-medium text-gray-600">
                            {positionFromCurrent}번째 전
                          </div>
                          <div className="text-3xl text-center flex items-center justify-center w-14 h-14">
                            {shape.symbol}
                          </div>
                        </div>
                      );
                    })}
                  </div>
                </div>
              )}

              <div className="flex-shrink-0">
                <div className="bg-gradient-to-br from-gray-50 to-gray-100 rounded-3xl shadow-2xl p-16 w-80 h-80 flex items-center justify-center relative">
                  {isShowingShape && currentShape && (
                    <div className="text-9xl flex items-center justify-center animate-pulse">
                      {currentShape.symbol}
                    </div>
                  )}
                  {isInInterval && (
                    <div className="text-gray-400 text-sm">준비 중...</div>
                  )}
                  {userAnswer && isShowingShape && (
                    <div className="absolute bottom-4 left-1/2 transform -translate-x-1/2 bg-green-500 text-white px-4 py-2 rounded-full text-sm font-medium shadow-lg">
                      ✓ 응답 완료
                    </div>
                  )}
                </div>
              </div>

              <div className="flex-shrink-0 space-y-3">
                <div className="text-center text-gray-700 font-medium mb-4 text-sm">
                  현재 도형이<br/>
                  {nBackLevel === 2 ? '2번째 전' : '2번째/3번째 전'} 도형과<br/>
                  같은지 판단하세요
                </div>

                <div className="bg-gray-50 rounded-lg p-3 flex items-center gap-3">
                  <div className="bg-white border-2 border-gray-300 rounded px-4 py-2 shadow-sm">
                    <span className="font-mono text-xs text-gray-600">SPACE</span>
                  </div>
                  <span className="text-sm text-gray-700">다름</span>
                </div>

                <div className="bg-blue-50 rounded-lg p-3 flex items-center gap-3">
                  <div className="bg-white border-2 border-blue-400 rounded px-3 py-2 shadow-sm">
                    <span className="text-xl text-blue-600">←</span>
                  </div>
                  <span className="text-sm text-blue-700 font-medium">2번째 전</span>
                </div>

                {nBackLevel === 3 && (
                  <div className="bg-purple-50 rounded-lg p-3 flex items-center gap-3">
                    <div className="bg-white border-2 border-purple-400 rounded px-3 py-2 shadow-sm">
                      <span className="text-xl text-purple-600">→</span>
                    </div>
                    <span className="text-sm text-purple-700 font-medium">3번째 전</span>
                  </div>
                )}

                <div className="mt-4 pt-4 border-t border-gray-200">
                  <div className="text-xs text-gray-500 mb-2">현재 점수</div>
                  <div className="text-2xl font-bold text-gray-800">
                    {score} / {totalTrials}
                  </div>
                  {currentTrial <= 2 && (
                    <div className="text-xs text-amber-600 mt-1">연습 문제</div>
                  )}
                </div>
              </div>
            </div>
          </div>
        )}

        {gameState === 'finished' && (
          <div className="bg-white rounded-2xl shadow-xl p-8 text-center">
            <h2 className="text-4xl font-bold mb-8 text-gray-800">게임 완료!</h2>
            
            <div className="grid grid-cols-2 gap-8 mb-10">
              <div className="bg-blue-50 rounded-xl p-6">
                <div className="text-sm text-blue-600 font-medium mb-2">최종 점수</div>
                <div className="text-4xl font-bold text-blue-700">{score} / {totalTrials}</div>
              </div>
              <div className="bg-green-50 rounded-xl p-6">
                <div className="text-sm text-green-600 font-medium mb-2">정답률</div>
                <div className="text-4xl font-bold text-green-700">
                  {totalTrials > 0 ? Math.round((score / totalTrials) * 100) : 0}%
                </div>
              </div>
            </div>

            <button
              onClick={resetGame}
              className="bg-gradient-to-r from-blue-600 to-blue-700 text-white py-4 px-12 rounded-xl text-xl font-bold hover:from-blue-700 hover:to-blue-800 transition-all shadow-lg"
            >
              다시 하기
            </button>
          </div>
        )}
      </div>
    </div>
  );
};
